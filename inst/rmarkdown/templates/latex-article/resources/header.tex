
% Make code a bit smaller
\ifdef{\DefineVerbatimEnvironment}{
  \DefineVerbatimEnvironment{Highlighting}{Verbatim}{
    commandchars=\\\{\},%
    fontsize=\small
  }
}{}

% Better title: https://tex.stackexchange.com/a/216121/5688
% - Bold font
\makeatletter
\def\@maketitle{%
  \newpage
  \null
  \vskip 2em%
  \let \footnote \thanks
    {\bfseries\LARGE \@title \par}%
    \vskip 1.5em%
    {\large
      \lineskip .5em%
      \@author\par}%
    {\large \@date}%
  \par
  \vskip 1.5em}
\makeatother
